    var a = 0; // function-scoped (or globally-scoped)
    if (true) {
        var a = 1; // reassigns outer x because of hoisting
        var b = 3; // reassigns outer y (gets "hoisted" to the top of scope)

        let d = 7; // block-scoped
        console.log(a, b, c, d); // Output: 1, 3, 3, 7
    }

}
